# cSpell:disable
cmake_minimum_required(VERSION 3.20)
project(BinaryFetch LANGUAGES CXX RC)

# Set project root
set(PROJECT_ROOT "${CMAKE_CURRENT_SOURCE_DIR}")
message(STATUS "Project root: ${PROJECT_ROOT}")

# C++ standard
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

# Output directories
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/bin")
set(CMAKE_ARCHIVE_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/lib")
set(CMAKE_LIBRARY_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/lib")

# Compiler settings for Clang on Windows
if(CMAKE_CXX_COMPILER_ID MATCHES "Clang")
    # Clang-specific flags for Windows compatibility
    add_compile_options(
        -fms-extensions
        -fms-compatibility
        -Wno-microsoft
        -Wno-unused-command-line-argument
    )
    
    # Link with Windows subsystems
    if(NOT CMAKE_BUILD_TYPE STREQUAL "Debug")
        add_link_options(/SUBSYSTEM:CONSOLE)
    endif()
endif()

# Collect source files
set(SOURCE_FILES
    ${PROJECT_ROOT}/src/main.cpp
    ${PROJECT_ROOT}/src/compact_disk_info.cpp
    ${PROJECT_ROOT}/src/CompactAudio.cpp
    ${PROJECT_ROOT}/src/CompactCPU.cpp
    ${PROJECT_ROOT}/src/CompactGPU.cpp
    ${PROJECT_ROOT}/src/CompactMemory.cpp
    ${PROJECT_ROOT}/src/CompactNetwork.cpp
    ${PROJECT_ROOT}/src/CompactOS.cpp
    ${PROJECT_ROOT}/src/CompactPerformance.cpp
    ${PROJECT_ROOT}/src/CompactScreen.cpp
    ${PROJECT_ROOT}/src/CompactSystem.cpp
    ${PROJECT_ROOT}/src/CompactUser.cpp
    ${PROJECT_ROOT}/src/ConfigReader.cpp
    ${PROJECT_ROOT}/src/ConsoleUtils.cpp
    ${PROJECT_ROOT}/src/CPUInfo.cpp
    ${PROJECT_ROOT}/src/DisplayInfo.cpp
    ${PROJECT_ROOT}/src/DtailedGPUInfo.cpp
    ${PROJECT_ROOT}/src/ExtraInfo.cpp
    ${PROJECT_ROOT}/src/GPUInfo.cpp
    ${PROJECT_ROOT}/src/Helpers.cpp
    ${PROJECT_ROOT}/src/MemoryInfo.cpp
    ${PROJECT_ROOT}/src/NetworkInfo.cpp
    ${PROJECT_ROOT}/src/OSInfo.cpp
    ${PROJECT_ROOT}/src/PerformanceInfo.cpp
    ${PROJECT_ROOT}/src/personalization_info.cpp
    ${PROJECT_ROOT}/src/StorageInfo.cpp
    ${PROJECT_ROOT}/src/SystemInfo.cpp
    ${PROJECT_ROOT}/src/TimeInfo.cpp
    ${PROJECT_ROOT}/src/UserInfo.cpp
    ${PROJECT_ROOT}/src/AsciiArt.cpp
)

# Header files
set(HEADER_FILES
    ${PROJECT_ROOT}/include/AsciiArt.h
    ${PROJECT_ROOT}/include/compact_disk_info.h
    ${PROJECT_ROOT}/include/CompactAudio.h
    ${PROJECT_ROOT}/include/CompactCPU.h
    ${PROJECT_ROOT}/include/CompactGPU.h
    ${PROJECT_ROOT}/include/CompactMemory.h
    ${PROJECT_ROOT}/include/CompactNetwork.h
    ${PROJECT_ROOT}/include/CompactOS.h
    ${PROJECT_ROOT}/include/CompactPerformance.h
    ${PROJECT_ROOT}/include/CompactScreen.h
    ${PROJECT_ROOT}/include/CompactSystem.h
    ${PROJECT_ROOT}/include/CompactUser.h
    ${PROJECT_ROOT}/include/ConfigReader.h
    ${PROJECT_ROOT}/include/CPUInfo.h
    ${PROJECT_ROOT}/include/DetailedGPUInfo.h
    ${PROJECT_ROOT}/include/DisplayInfo.h
    ${PROJECT_ROOT}/include/ExtraInfo.h
    ${PROJECT_ROOT}/include/GPUInfo.h
    ${PROJECT_ROOT}/include/MemoryInfo.h
    ${PROJECT_ROOT}/include/NetworkInfo.h
    ${PROJECT_ROOT}/include/OSInfo.h
    ${PROJECT_ROOT}/include/PerformanceInfo.h
    ${PROJECT_ROOT}/include/personalization_info.h
    ${PROJECT_ROOT}/include/StorageInfo.h
    ${PROJECT_ROOT}/include/SystemInfo.h
    ${PROJECT_ROOT}/include/TimeInfo.h
    ${PROJECT_ROOT}/include/UserInfo.h
    ${PROJECT_ROOT}/include/nvapi.h
    ${PROJECT_ROOT}/include/json.hpp
)

# Resource files
set(RESOURCE_FILES
    ${PROJECT_ROOT}/binary_fetch_v1/binary_fetch_v1.rc
    ${PROJECT_ROOT}/binary_fetch_v1/resources.rc
    ${PROJECT_ROOT}/binary_fetch_v1/version_info.rc
)

# Create executable
add_executable(${PROJECT_NAME} WIN32
    ${SOURCE_FILES}
    ${HEADER_FILES}
    ${RESOURCE_FILES}
)

# Include directories
target_include_directories(${PROJECT_NAME} PRIVATE
    ${PROJECT_ROOT}/include
    ${PROJECT_ROOT}/binary_fetch_v1/include
    ${PROJECT_ROOT}/binary_fetch_v1/include/Nvidia
    ${PROJECT_ROOT}/binary_fetch_v1/nlohmann
)

# Preprocessor definitions
target_compile_definitions(${PROJECT_NAME} PRIVATE
    UNICODE
    _UNICODE
    WIN32_LEAN_AND_MEAN
    NOMINMAX
    $<$<CONFIG:Debug>:_DEBUG>
    $<$<CONFIG:Release>:NDEBUG>
)

# Compiler options
target_compile_options(${PROJECT_NAME} PRIVATE
    $<$<CXX_COMPILER_ID:MSVC>:
        /W4
        /EHsc
        /MP
    >
    $<$<CXX_COMPILER_ID:Clang>:
        -Wall
        -Wextra
        -Wpedantic
        -Wno-microsoft-cast
        -Wno-microsoft-include
    >
)

# Link directories for NVAPI (use relative paths)
target_link_directories(${PROJECT_NAME} PRIVATE
    ${PROJECT_ROOT}/binary_fetch_v1/lib
)

# Link libraries (without .lib extension for CMake compatibility)
target_link_libraries(${PROJECT_NAME} PRIVATE
    # NVAPI
    nvapi64
    
    # Windows system libraries
    kernel32
    user32
    gdi32
    winspool
    comdlg32
    advapi32
    shell32
    ole32
    oleaut32
    uuid
    odbc32
    odbccp32
    
    # Additional Windows libraries
    iphlpapi    # For network information
    pdh         # For performance data helper
    ws2_32      # For Winsock (networking)
    setupapi    # For device information
    version     # For version info
)

# Copy NVAPI DLL to output directory (post-build step)
add_custom_command(TARGET ${PROJECT_NAME} POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy_if_different
        "${PROJECT_ROOT}/binary_fetch_v1/lib/nvapi64.dll"
        $<TARGET_FILE_DIR:${PROJECT_NAME}>
    COMMENT "Copying NVAPI DLL to output directory"
)

# Source groups for VS Code organization
source_group(TREE ${PROJECT_ROOT}/src PREFIX "Source Files" FILES ${SOURCE_FILES})
source_group(TREE ${PROJECT_ROOT}/include PREFIX "Header Files" FILES ${HEADER_FILES})
source_group("Resource Files" FILES ${RESOURCE_FILES})

# Installation (optional)
install(TARGETS ${PROJECT_NAME}
    RUNTIME DESTINATION bin
    LIBRARY DESTINATION lib
    ARCHIVE DESTINATION lib
)

# Status messages
message(STATUS "BinaryFetch CMake configuration complete")
message(STATUS "Source files count: ${CMAKE_ARGC}")
message(STATUS "Header files count: ${CMAKE_ARGC}")
message(STATUS "Using NVAPI from: ${PROJECT_ROOT}/binary_fetch_v1/lib")
